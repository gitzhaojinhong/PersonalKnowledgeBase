## 七、高级用法与实战

### 7.1 自定义 SQL（注解 + XML）

MyBatis-Plus 完全兼容 MyBatis 的自定义 SQL 方式。

#### 7.1.1 注解方式

```java
public interface UserMapper extends BaseMapper<User> {
    
    // 简单查询
    @Select("SELECT * FROM t_user WHERE age > #{age}")
    List<User> selectByAge(@Param("age") Integer age);
    
    // 关联查询
    @Select("""
        SELECT u.*, d.dept_name 
        FROM t_user u 
        LEFT JOIN t_dept d ON u.dept_id = d.id 
        WHERE u.id = #{userId}
        """)
    @Results({
        @Result(property = "id", column = "id"),
        @Result(property = "username", column = "username"),
        @Result(property = "deptName", column = "dept_name")
    })
    User selectUserWithDept(@Param("userId") Long userId);
}
```

#### 7.1.2 XML 方式

```xml
<!-- resources/mapper/UserMapper.xml -->
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" 
    "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.example.mapper.UserMapper">
    
    <resultMap id="userResultMap" type="com.example.entity.User">
        <id property="id" column="id"/>
        <result property="username" column="user_name"/>
        <result property="createTime" column="create_time"/>
    </resultMap>
    
    <select id="selectUserDetail" resultMap="userResultMap">
        SELECT * FROM t_user
        WHERE deleted = 0
        <if test="username != null and username != ''">
            AND user_name LIKE CONCAT('%', #{username}, '%')
        </if>
        <if test="minAge != null">
            AND age >= #{minAge}
        </if>
        ORDER BY create_time DESC
    </select>
    
</mapper>
```

#### 7.1.3 Mapper 接口声明

```java
public interface UserMapper extends BaseMapper<User> {
    
    List<User> selectUserDetail(@Param("username") String username, 
                                 @Param("minAge") Integer minAge);
}
```

### 7.2 关联查询方案

MyBatis-Plus 本身不直接支持多表关联，推荐以下方案：

#### 7.2.1 方案一：自定义 SQL（推荐）

```xml
<select id="selectUserWithOrders" resultMap="userWithOrdersMap">
    SELECT u.*, o.id as order_id, o.amount, o.create_time as order_time
    FROM t_user u
    LEFT JOIN t_order o ON u.id = o.user_id
    WHERE u.deleted = 0 AND u.id = #{userId}
</select>

<resultMap id="userWithOrdersMap" type="com.example.vo.UserVO">
    <id property="id" column="id"/>
    <result property="username" column="username"/>
    <collection property="orders" ofType="com.example.vo.OrderVO">
        <id property="id" column="order_id"/>
        <result property="amount" column="amount"/>
        <result property="createTime" column="order_time"/>
    </collection>
</resultMap>
```

#### 7.2.2 方案二：单表查询 + 内存组装

```java
// 1. 查询用户
User user = userService.getById(userId);

// 2. 查询订单列表
List<Order> orders = orderService.list(
    Wrappers.<Order>lambdaQuery().eq(Order::getUserId, userId)
);

// 3. 组装 VO
UserVO vo = new UserVO();
BeanUtils.copyProperties(user, vo);
vo.setOrders(orders);
```

> **方案对比**：方案一适合单次查询数据量大的场景；方案二适合微服务拆分、需要分别查询不同服务的场景。

### 7.3 常见业务场景实战

#### 7.3.1 用户注册（存在则更新，不存在则插入）

```java
public boolean register(UserDTO dto) {
    User user = new User();
    BeanUtils.copyProperties(dto, user);
    
    // 根据手机号判断用户是否存在
    User exist = userService.lambdaQuery()
        .eq(User::getPhone, dto.getPhone())
        .one();
    
    if (exist != null) {
        // 用户已存在，更新信息
        user.setId(exist.getId());
        return userService.updateById(user);
    }
    
    // 新用户，设置默认密码和状态
    user.setPassword(passwordEncoder.encode(dto.getPassword()));
    user.setStatus(UserStatus.ENABLED);
    return userService.save(user);
}
```

#### 7.3.2 数据导入（批量 SaveOrUpdate）

```java
public void importUsers(List<UserImportDTO> dtoList) {
    List<User> userList = dtoList.stream().map(dto -> {
        User user = new User();
        BeanUtils.copyProperties(dto, user);
        // 设置主键（用于判断更新还是插入）
        if (dto.getId() != null) {
            user.setId(dto.getId());
        }
        return user;
    }).collect(Collectors.toList());
    
    // 批量保存或更新（每 500 条一批）
    userService.saveOrUpdateBatch(userList, 500);
}
```

#### 7.3.3 分页 + 复杂条件 + 排序

```java
public PageResult<UserVO> searchUserPage(UserSearchParam param) {
    // 构建分页对象
    Page<User> page = new Page<>(param.getPageNum(), param.getPageSize());
    
    // 构建查询条件
    LambdaQueryWrapper<User> wrapper = Wrappers.lambdaQuery();
    wrapper.like(StringUtils.isNotBlank(param.getUsername()), 
                 User::getUsername, param.getUsername())
           .ge(param.getStartTime() != null, 
               User::getCreateTime, param.getStartTime())
           .le(param.getEndTime() != null, 
               User::getCreateTime, param.getEndTime())
           .eq(param.getStatus() != null, 
               User::getStatus, param.getStatus())
           .orderByDesc(User::getCreateTime);
    
    // 执行分页查询
    Page<User> result = userService.page(page, wrapper);
    
    // 转换为 VO
    List<UserVO> voList = result.getRecords().stream().map(user -> {
        UserVO vo = new UserVO();
        BeanUtils.copyProperties(user, vo);
        return vo;
    }).collect(Collectors.toList());
    
    return new PageResult<>(voList, result.getTotal(), 
                           result.getCurrent(), result.getPages());
}
```

#### 7.3.4 Service 层事务管理

```java
@Service
public class OrderServiceImpl extends ServiceImpl<OrderMapper, Order> implements IOrderService {
    
    @Autowired
    private ProductService productService;
    
    @Autowired
    private OrderItemService orderItemService;
    
    @Override
    @Transactional(rollbackFor = Exception.class)//添加回滚逻辑：所有异常都回滚（最安全）
    public void createOrder(OrderDTO dto) {
        // 1. 扣减库存
        boolean success = productService.deductStock(dto.getProductId(), dto.getQuantity());
        if (!success) {
            throw new BusinessException("库存不足");
        }
        
        // 2. 创建订单
        Order order = new Order();
        BeanUtils.copyProperties(dto, order);
        order.setOrderNo(generateOrderNo());
        order.setStatus(OrderStatus.PENDING);
        this.save(order);
        
        // 3. 创建订单明细
        List<OrderItem> items = dto.getItems().stream().map(itemDTO -> {
            OrderItem item = new OrderItem();
            BeanUtils.copyProperties(itemDTO, item);
            item.setOrderId(order.getId());
            return item;
        }).collect(Collectors.toList());
        orderItemService.saveBatch(items);
        
        // 4. 记录日志（可选）
        // ...
    }
}
```

#### 7.3.5 树形结构查询（递归）

```java
@Data
@TableName("t_department")
public class Department {
    @TableId(type = IdType.AUTO)
    private Long id;
    private String name;
    private Long parentId;  // 父部门ID，0表示根节点
    private Integer sort;
}

@Service
public class DepartmentServiceImpl extends ServiceImpl<DepartmentMapper, Department> 
        implements IDepartmentService {

    /**
     * 获取部门树形结构（对外接口）
     */
    @Override
    public List<DepartmentVO> getTree() {
        // 1. 查询【所有】部门数据，并按 sort 升序排列
        List<Department> allDeptList = this.list(
            Wrappers.<Department>lambdaQuery().orderByAsc(Department::getSort)
        );

        // 2. 从 parentId = 0 开始，递归构建树形结构
        return buildTree(allDeptList, 0L);
    }

    /**
     * 【核心递归方法】构建部门树
     * @param all 所有部门列表（只查一次DB，性能高）
     * @param parentId 父级ID（开始传0）
     * @return 树形结构列表
     */
    private List<DepartmentVO> buildTree(List<Department> all, Long parentId) {
        return all.stream()
            // 过滤：只保留当前父ID的子部门
            .filter(dept -> dept.getParentId().equals(parentId))
            
            // 转换：Department → DepartmentVO
            .map(dept -> {
                DepartmentVO vo = new DepartmentVO();
                BeanUtils.copyProperties(dept, vo); // 属性拷贝
                
                // 【递归】找当前部门的子部门
                vo.setChildren(buildTree(all, dept.getId()));//将子部门放到children属性中
                return vo;
            })
            
            .collect(Collectors.toList());
    }
}
```
