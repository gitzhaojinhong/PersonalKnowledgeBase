## 八、扩展功能

### 8.1 代码生成器（3.5.1 以下版本）

**AutoGenerator** 是 MyBatis-Plus 的传统代码生成器，适用于 3.5.1 以下版本。

#### 8.1.1 添加依赖

```xml
<dependency>
    <groupId>com.baomidou</groupId>
    <artifactId>mybatis-plus-generator</artifactId>
    <version>3.5.0</version>
</dependency>

<!-- 模板引擎（三选一） -->
<dependency>
    <groupId>org.apache.velocity</groupId>
    <artifactId>velocity-engine-core</artifactId>
    <version>2.3</version>
</dependency>
```

#### 8.1.2 完整生成示例

```java
import com.baomidou.mybatisplus.generator.AutoGenerator;
import com.baomidou.mybatisplus.generator.config.*;
import com.baomidou.mybatisplus.generator.config.rules.NamingStrategy;
import com.baomidou.mybatisplus.generator.engine.VelocityTemplateEngine;

public class CodeGenerator {
    
    public static void main(String[] args) {
        AutoGenerator mpg = new AutoGenerator();
        
        // ========== 全局配置 ==========
        GlobalConfig gc = new GlobalConfig();
        String projectPath = System.getProperty("user.dir");
        gc.setOutputDir(projectPath + "/src/main/java");
        gc.setAuthor("yourname");
        gc.setOpen(false);  // 生成后是否打开目录
        gc.setSwagger2(true);  // 实体属性 Swagger2 注解
        gc.setServiceName("%sService");  // Service 接口名去除前缀 I
        mpg.setGlobalConfig(gc);
        
        // ========== 数据源配置 ==========
        DataSourceConfig dsc = new DataSourceConfig();
        dsc.setUrl("jdbc:mysql://localhost:3306/mydb?useUnicode=true&characterEncoding=utf8");
        dsc.setDriverName("com.mysql.cj.jdbc.Driver");
        dsc.setUsername("root");
        dsc.setPassword("password");
        mpg.setDataSource(dsc);
        
        // ========== 包配置 ==========
        PackageConfig pc = new PackageConfig();
        pc.setParent("com.example");
        pc.setModuleName("system");
        pc.setEntity("entity");
        pc.setMapper("mapper");
        pc.setService("service");
        pc.setServiceImpl("service.impl");
        pc.setController("controller");
        mpg.setPackageInfo(pc);
        
        // ========== 策略配置 ==========
        StrategyConfig strategy = new StrategyConfig();
        strategy.setNaming(NamingStrategy.underline_to_camel);      // 表名转驼峰
        strategy.setColumnNaming(NamingStrategy.underline_to_camel); // 字段名转驼峰
        strategy.setEntityLombokModel(true);                         // Lombok 模式
        strategy.setRestControllerStyle(true);                       // RestController 风格
        strategy.setControllerMappingHyphenStyle(true);              // URL 驼峰转连字符
        strategy.setTablePrefix("t_");                               // 表前缀
        strategy.setInclude("t_user", "t_order", "t_product");       // 指定生成表
        strategy.setLogicDeleteFieldName("deleted");                 // 逻辑删除字段
        
        // 自动填充字段配置
        ArrayList<TableFill> tableFills = new ArrayList<>();
        tableFills.add(new TableFill("create_time", FieldFill.INSERT));
        tableFills.add(new TableFill("update_time", FieldFill.INSERT_UPDATE));
        strategy.setTableFillList(tableFills);
        
        mpg.setStrategy(strategy);
        mpg.setTemplateEngine(new VelocityTemplateEngine());
        mpg.execute();
    }
}
```

### 8.2 新版代码生成器（3.5.1+）

**FastAutoGenerator** 是 3.5.1+ 版本推出的新版代码生成器，采用流式 API，更加简洁强大。

#### 8.2.1 添加依赖

```xml
<dependency>
    <groupId>com.baomidou</groupId>
    <artifactId>mybatis-plus-generator</artifactId>
    <version>3.5.7</version>
</dependency>
<dependency>
    <groupId>org.freemarker</groupId>
    <artifactId>freemarker</artifactId>
</dependency>
```

#### 8.2.2 完整生成示例

```java
import com.baomidou.mybatisplus.generator.FastAutoGenerator;
import com.baomidou.mybatisplus.generator.config.OutputFile;
import com.baomidou.mybatisplus.generator.config.rules.DbColumnType;
import com.baomidou.mybatisplus.generator.engine.FreemarkerTemplateEngine;
import java.sql.Types;
import java.util.Collections;

public class NewCodeGenerator {
    
    public static void main(String[] args) {
        FastAutoGenerator.create(
                "jdbc:mysql://localhost:3306/mydb?serverTimezone=Asia/Shanghai",
                "root",
                "password"
            )
            // ========== 全局配置 ==========
            .globalConfig(builder -> {
                builder.author("yourname")
                       .enableSwagger()           // 开启 Swagger 模式
                       .outputDir(System.getProperty("user.dir") + "/src/main/java")
                       .disableOpenDir();         // 生成后不打开目录
            })
            // ========== 包配置 ==========
            .packageConfig(builder -> {
                builder.parent("com.example")
                       .moduleName("system")
                       .entity("entity")
                       .service("service")
                       .serviceImpl("service.impl")
                       .mapper("mapper")
                       .xml("mapper")
                       .controller("controller")
                       .pathInfo(Collections.singletonMap(
                           OutputFile.xml, 
                           System.getProperty("user.dir") + "/src/main/resources/mapper"
                       ));
            })
            // ========== 策略配置 ==========
            .strategyConfig(builder -> {
                builder.addInclude("t_user", "t_order", "t_product")  // 指定表
                       .addTablePrefix("t_")                            // 过滤表前缀
                       
                       // Entity 策略
                       .entityBuilder()
                       .enableLombok()                                  // Lombok
                       .enableChainModel()                              // 链式模型
                       .enableTableFieldAnnotation()                    // 生成字段注解
                       .logicDeleteColumnName("deleted")                // 逻辑删除字段
                       .addTableFills(
                           new Column("create_time", FieldFill.INSERT),
                           new Column("update_time", FieldFill.INSERT_UPDATE)
                       )
                       .naming(NamingStrategy.underline_to_camel)
                       .columnNaming(NamingStrategy.underline_to_camel)
                       
                       // Controller 策略
                       .controllerBuilder()
                       .enableRestStyle()                               // RestController
                       .enableHyphenStyle()                             // URL 驼峰转连字符
                       
                       // Service 策略
                       .serviceBuilder()
                       .formatServiceFileName("%sService")
                       .formatServiceImplFileName("%sServiceImpl")
                       
                       // Mapper 策略
                       .mapperBuilder()
                       .enableMapperAnnotation()                        // @Mapper 注解
                       .enableBaseResultMap()                           // 生成 ResultMap
                       .enableBaseColumnList();                         // 生成 BaseColumnList
            })
            // ========== 模板引擎 ==========
            .templateEngine(new FreemarkerTemplateEngine())
            .execute();
    }
}
```

### 8.3 SQL 注入器（自定义 Mapper 方法）

当 BaseMapper 提供的方法无法满足需求时，可通过 **SQL 注入器**扩展自定义方法。

#### 8.3.1 内置选装件方法

MyBatis-Plus 提供了一些额外的选装方法，需要手动注入：

| 方法                         | 说明                                   | 所在包                                  |
| ---------------------------- | -------------------------------------- | --------------------------------------- |
| `insertBatchSomeColumn`      | 批量插入指定字段（排除逻辑删除等字段） | `extension.injector.methods.additional` |
| `alwaysUpdateSomeColumnById` | 根据 ID 始终更新某些字段               | `extension.injector.methods.additional` |
| `logicDeleteByIdWithFill`    | 逻辑删除并填充字段                     | `extension.injector.methods.additional` |

**使用示例**：

```java
// 在 Mapper 接口中使用注入的选装件方法
public interface UserMapper extends BaseMapper<User> {
    // 批量插入（自动排除逻辑删除字段）
    int insertBatchSomeColumn(List<User> entityList);

    // 始终更新某些字段（如 update_time）
    int alwaysUpdateSomeColumnById(@Param("et") User entity);
}
```

```java
// 业务层调用
// 1. 批量插入（insertBatchSomeColumn 会自动排除 deleted、create_time 等字段）
List<User> userList = Arrays.asList(user1, user2, user3);
userMapper.insertBatchSomeColumn(userList);

// 2. 始终更新某些字段（即使 entity 中这些字段为 null，也会强制更新）
User updateUser = new User();
updateUser.setId(1L);
updateUser.setUsername("新名字");  // 只会更新 username 和 update_time
userMapper.alwaysUpdateSomeColumnById(updateUser);
```

#### 8.3.2 注入选装件

```java
@Bean
public DefaultSqlInjector sqlInjector() {
    return new DefaultSqlInjector() {
        @Override
        public List<AbstractMethod> getMethodList(Class<?> mapperClass, TableInfo tableInfo) {
            List<AbstractMethod> methodList = super.getMethodList(mapperClass, tableInfo);
            // 批量插入（排除逻辑删除字段）
            methodList.add(new InsertBatchSomeColumn(
                i -> !i.getColumn().equals("deleted") && !i.getColumn().equals("create_time")
            ));
            // 始终更新某些字段（如 update_time）
            methodList.add(new AlwaysUpdateSomeColumnById(
                i -> i.getColumn().equals("update_time")
            ));
            return methodList;
        }
    };
}
```

#### 8.3.3 自定义注入方法

**Step 1：定义方法类**

```java
public class DeleteAll extends AbstractMethod {
    
    @Override
    public MappedStatement injectMappedStatement(Class<?> mapperClass, Class<?> modelClass, 
                                                   TableInfo tableInfo) {
        // SQL 模板
        String sql = "DELETE FROM " + tableInfo.getTableName();
        SqlSource sqlSource = languageDriver.createSqlSource(configuration, sql, modelClass);
        
        return this.addDeleteMappedStatement(mapperClass, "deleteAll", sqlSource);
    }
}
```

**Step 2：注入到容器**

```java
@Bean
public DefaultSqlInjector sqlInjector() {
    return new DefaultSqlInjector() {
        @Override
        public List<AbstractMethod> getMethodList(Class<?> mapperClass, TableInfo tableInfo) {
            List<AbstractMethod> methods = super.getMethodList(mapperClass, tableInfo);
            methods.add(new DeleteAll());
            return methods;
        }
    };
}
```

**Step 3：在 Mapper 中声明**

```java
public interface UserMapper extends BaseMapper<User> {
    // 自定义方法声明
    int deleteAll();
}
```

### 8.4 ActiveRecord 模式

**ActiveRecord（活动记录）**模式让实体类自身具备 CRUD 能力，无需依赖 Mapper 或 Service。

#### 8.4.1 实体类继承 Model

```java
import com.baomidou.mybatisplus.extension.activerecord.Model;

@Data
@TableName("t_user")
public class User extends Model<User> {
    
    @TableId(type = IdType.AUTO)
    private Long id;
    private String username;
    private Integer age;
    private String email;
    
    @Override
    protected Serializable pkVal() {
        return this.id;  // 指定主键值
    }
}
```

#### 8.4.2 直接使用实体 CRUD

```java
// 插入
User user = new User();
user.setUsername("ActiveRecord");
user.setAge(20);
user.insert();

// 查询
User query = new User();
query.setId(1L);
User result = query.selectById();

// 查询所有
List<User> all = query.selectAll();

// 更新
user.setAge(25);
user.updateById();

// 删除
user.deleteById();

// 条件查询
List<User> list = user.selectList(
    Wrappers.<User>lambdaQuery().gt(User::getAge, 18)
);

// 条件统计
long count = query.selectCount(
    Wrappers.<User>lambdaQuery().eq(User::getStatus, 1)
);
```

> **适用场景**：简单 CRUD、脚本工具、快速原型开发；**不推荐**在复杂业务系统中大量使用。

### 8.5 SimpleQuery / Db Kit

#### 8.5.1 SimpleQuery

**SimpleQuery** 是对 `selectList` 结果的 Stream 流式处理封装。

```java
import static com.baomidou.mybatisplus.extension.toolkit.SimpleQuery;

// 1. 转换为 Map（key 为指定字段，value 为实体）
Map<Long, User> userMap = SimpleQuery.keyMap(
    Wrappers.<User>lambdaQuery().gt(User::getAge, 18),
    User::getId
);

// 2. 分组（按年龄分组）
Map<Integer, List<User>> groupByAge = SimpleQuery.groupBy(
    Wrappers.<User>lambdaQuery(),
    User::getAge
);

// 3. 提取某一列的 List
List<String> names = SimpleQuery.list(
    Wrappers.<User>lambdaQuery().eq(User::getStatus, 1),
    User::getUsername
);
```

#### 8.5.2 Db Kit（静态工具类）

**Db Kit** 提供静态方法直接执行 CRUD，避免 Service 循环注入问题。

```java
import com.baomidou.mybatisplus.extension.toolkit.Db;

// 查询
User user = Db.getById(1L, User.class);
List<User> list = Db.list(Wrappers.<User>lambdaQuery().eq(User::getStatus, 1));

// 插入
Db.save(user);
Db.saveBatch(userList);

// 更新
Db.updateById(user);

// 删除
Db.removeById(1L, User.class);

// 统计
long count = Db.count(Wrappers.<User>lambdaQuery());
```

**Db.lambdaQuery / Db.lambdaUpdate（推荐）**：

```java
// 静态 Lambda 查询（无需手动创建 Wrapper）
List<User> users = Db.lambdaQuery(User.class)
    .eq(User::getStatus, 1)
    .like(User::getUsername, "张")
    .list();

// 静态 Lambda 更新
Db.lambdaUpdate(User.class)
    .set(User::getStatus, 0)
    .lt(User::getLastLoginTime, LocalDateTime.now().minusDays(30))
    .update();

// 静态 Lambda 删除
Db.lambdaUpdate(User.class)
    .eq(User::getStatus, 0)
    .remove();
```

> **注意**：Db Kit 方法需要确保对应实体类已注册到 MyBatis-Plus 的表信息缓存中。

### 8.6 多数据源

MyBatis-Plus 生态提供了 **dynamic-datasource** 组件实现多数据源切换。

#### 8.6.1 添加依赖

```xml
<!-- Spring Boot 2.x -->
<dependency>
    <groupId>com.baomidou</groupId>
    <artifactId>dynamic-datasource-spring-boot-starter</artifactId>
    <version>4.3.1</version>
</dependency>

<!-- Spring Boot 3.x -->
<dependency>
    <groupId>com.baomidou</groupId>
    <artifactId>dynamic-datasource-spring-boot3-starter</artifactId>
    <version>4.3.1</version>
</dependency>
```

#### 8.6.2 YAML 配置

```yaml
spring:
  datasource:
    dynamic:
      primary: master          # 默认数据源
      strict: false            # 严格模式（未找到数据源时是否抛异常）
      datasource:
        master:
          url: jdbc:mysql://localhost:3306/db_master?serverTimezone=Asia/Shanghai
          username: root
          password: root
          driver-class-name: com.mysql.cj.jdbc.Driver
        slave:
          url: jdbc:mysql://localhost:3307/db_slave?serverTimezone=Asia/Shanghai
          username: root
          password: root
          driver-class-name: com.mysql.cj.jdbc.Driver
```

#### 8.6.3 注解切换数据源

```java
@Service
@DS("slave")  // 类级别：默认使用 slave 数据源
public class UserServiceImpl implements IUserService {
    
    @Autowired
    private UserMapper userMapper;
    
    @Override
    @DS("master")  // 方法级别：覆盖类级别，使用 master
    public boolean save(User user) {
        return userMapper.insert(user) > 0;
    }
    
    @Override
    public List<User> list() {
        // 使用 slave 数据源（继承类级别配置）
        return userMapper.selectList(null);
    }
}
```

#### 8.6.4 注意事项

- **优先级**：方法注解 > 类注解 > 默认数据源
- **事务限制**：多数据源下，同一事务内只能操作一个数据源，跨数据源事务需借助 Seata 等分布式事务方案
- **数据源组**：支持下划线分组，如 `slave_1`、`slave_2` 属于 `slave` 组，切换 `@DS("slave")` 时会负载均衡
